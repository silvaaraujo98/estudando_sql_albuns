-- Obter os Generos mais vendidos por Pa√≠s

SELECT d.name,
       c.billingcountry,
       COUNT(c.invoiceid) AS sales_amount
FROM invoiceline AS a
INNER JOIN track AS b ON a.trackid = b.trackid
INNER JOIN invoice AS c ON a.invoiceid = c.invoiceid
INNER JOIN genre AS d ON b.genreid = d.genreid
GROUP BY d.name,
         c.billingcountry
ORDER BY c.billingcountry,
         sales_amount DESC;
